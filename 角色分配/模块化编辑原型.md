import React, { useState } from 'react';
import { 
  FileText, 
  Settings, 
  AlertTriangle, 
  Shield, 
  GripVertical, 
  Hash, 
  Link as LinkIcon, 
  ChevronRight, 
  ChevronDown,
  Layout,
  User,
  Activity,
  GitGraph // [新增] 引入新模块的图标
} from 'lucide-react';

// 1. [新增] 模拟初始数据中增加一种新类型 'process_link'
const initialData = [
  {
    id: 'DOC-100',
    type: 'heading',
    level: 1,
    content: '差旅费用报销管理制度',
    meta: { author: '财务部' }
  },
  {
    id: 'CLAUSE-101',
    type: 'clause',
    number: '1.0',
    content: '为了规范公司差旅费用的开支，特制定本制度。',
    meta: { status: 'effective' }
  },
  {
    id: 'CLAUSE-102',
    type: 'heading',
    level: 2,
    content: '报销原则',
    meta: {}
  },
  {
    id: 'RULE-201',
    type: 'rule',
    subtype: 'mandatory', 
    number: '2.1',
    content: '所有报销凭证必须为合法有效的增值税发票，且抬头必须为公司全称。',
    meta: { 
      enforcement: 'strict', 
      owner: '财务总监',
      backlinks: 3 
    }
  },
  {
    id: 'LINK-401', // [新增] 这是一个新模块数据
    type: 'process_link', 
    number: '2.2',
    content: '关联主流程：差旅报销审批流 (Standard-01)',
    meta: { 
      targetProcessId: 'BPMN-FIN-01',
      version: '2.0'
    }
  },
  {
    id: 'RISK-305',
    type: 'risk',
    number: '2.3',
    content: '若发票丢失，需提供第三方证明并填写《遗失说明书》，单笔金额不得超过 500 元。',
    meta: { 
      riskLevel: 'medium', 
      control: '部门经理审批',
      backlinks: 1
    }
  },
  {
    id: 'CLAUSE-103',
    type: 'clause',
    number: '2.4',
    content: '出差人员应在返回公司后 5 个工作日内完成报销流程。',
    meta: { status: 'draft' }
  }
];

export default function StructuredEditor() {
  const [blocks, setBlocks] = useState(initialData);
  const [selectedId, setSelectedId] = useState('LINK-401'); // 默认选中新模块方便查看
  const [leftSidebarOpen, setLeftSidebarOpen] = useState(true);

  const selectedBlock = blocks.find(b => b.id === selectedId);

  const updateBlock = (id, field, value) => {
    setBlocks(blocks.map(b => 
      b.id === id ? { ...b, [field]: value } : b
    ));
  };

  // [新增] 简单的添加模块功能函数
  const addBlock = (type) => {
    const newBlock = {
      id: `NEW-${Math.floor(Math.random() * 1000)}`,
      type: type,
      number: 'X.X',
      content: type === 'process_link' ? '点击关联流程...' : '新插入的条款内容...',
      meta: {}
    };
    // 简单插入到最后，实际逻辑应该是插入到当前选中项后面
    setBlocks([...blocks, newBlock]);
    setSelectedId(newBlock.id);
  };

  // 2. [新增] 渲染左侧大纲树节点的逻辑
  const renderTreeNode = (block) => {
    const isSelected = block.id === selectedId;
    let icon = <div className="w-2 h-2 rounded-full bg-slate-300" />;
    
    if (block.type === 'heading') icon = <FileText size={14} className="text-slate-500" />;
    if (block.type === 'risk') icon = <AlertTriangle size={14} className="text-amber-500" />;
    if (block.type === 'rule') icon = <Shield size={14} className="text-red-500" />;
    // [新增] 树节点图标
    if (block.type === 'process_link') icon = <GitGraph size={14} className="text-purple-500" />;

    return (
      <div 
        key={block.id}
        onClick={() => setSelectedId(block.id)}
        className={`flex items-center gap-2 px-3 py-1.5 text-sm cursor-pointer rounded transition-colors ${
          isSelected ? 'bg-blue-50 text-blue-700 font-medium' : 'text-slate-600 hover:bg-slate-100'
        }`}
        style={{ paddingLeft: block.type === 'heading' ? '0.75rem' : '2rem' }}
      >
        {icon}
        <span className="truncate">{block.type === 'heading' ? block.content : block.id}</span>
      </div>
    );
  };

  return (
    <div className="flex h-screen w-full bg-slate-50 text-slate-800 font-sans overflow-hidden">
      
      {/* 1. 左侧栏 */}
      <div className={`${leftSidebarOpen ? 'w-64' : 'w-12'} flex-shrink-0 bg-white border-r border-slate-200 transition-all duration-300 flex flex-col`}>
        <div className="h-14 border-b border-slate-100 flex items-center px-4 justify-between">
          {leftSidebarOpen && <span className="font-semibold text-slate-700">文档结构</span>}
          <button onClick={() => setLeftSidebarOpen(!leftSidebarOpen)} className="p-1 hover:bg-slate-100 rounded text-slate-400">
            <Layout size={18} />
          </button>
        </div>
        
        {leftSidebarOpen && (
          <div className="flex-1 overflow-y-auto p-2 space-y-1">
            {blocks.map(renderTreeNode)}
          </div>
        )}
      </div>

      {/* 2. 中间栏 */}
      <div className="flex-1 flex flex-col min-w-0 bg-slate-50">
        {/* 工具栏 */}
        <div className="h-14 bg-white border-b border-slate-200 flex items-center px-6 gap-4 shadow-sm z-10">
          <div className="flex items-center gap-2 text-slate-500 text-sm border-r pr-4">
            <span className="font-bold text-slate-800">V1.0</span>
            <span>最后编辑于 刚刚</span>
          </div>
          <div className="flex gap-2">
            <button onClick={() => addBlock('clause')} className="p-2 hover:bg-slate-100 rounded text-slate-600" title="插入条款">+ 条款</button>
            <button onClick={() => addBlock('risk')} className="p-2 hover:bg-slate-100 rounded text-slate-600 flex items-center gap-1" title="插入风险点">
              <AlertTriangle size={16} className="text-amber-500"/> 风险点
            </button>
            <button onClick={() => addBlock('rule')} className="p-2 hover:bg-slate-100 rounded text-slate-600 flex items-center gap-1" title="插入控制规则">
              <Shield size={16} className="text-red-500"/> 强控规则
            </button>
            {/* [新增] 工具栏按钮 */}
            <button onClick={() => addBlock('process_link')} className="p-2 hover:bg-slate-100 rounded text-slate-600 flex items-center gap-1" title="插入流程挂载">
              <GitGraph size={16} className="text-purple-500"/> 流程挂载
            </button>
          </div>
        </div>

        {/* 编辑区域 */}
        <div className="flex-1 overflow-y-auto p-8 flex justify-center">
          <div className="w-full max-w-3xl bg-white shadow-sm min-h-[800px] p-12 rounded-lg border border-slate-200">
            
            {blocks.map((block) => {
              const isSelected = selectedId === block.id;
              
              let containerClass = "group relative mb-4 transition-all duration-200";
              let contentClass = "outline-none p-2 rounded border border-transparent";
              let sideBorder = null;

              if (isSelected) {
                contentClass += " bg-blue-50/30 border-blue-200";
              } else {
                contentClass += " hover:bg-slate-50";
              }

              if (block.type === 'heading') {
                contentClass += block.level === 1 ? " text-3xl font-bold mb-6 mt-2" : " text-xl font-bold mt-4 mb-2";
              } else if (block.type === 'risk') {
                contentClass += " bg-amber-50/50 text-slate-800";
                sideBorder = <div className="absolute left-0 top-0 bottom-0 w-1 bg-amber-400 rounded-l" />;
              } else if (block.type === 'rule') {
                contentClass += " bg-red-50/50 text-slate-800";
                sideBorder = <div className="absolute left-0 top-0 bottom-0 w-1 bg-red-500 rounded-l" />;
              } else if (block.type === 'process_link') {
                // [新增] 新模块的编辑器内样式
                contentClass += " bg-purple-50 text-purple-900 font-medium flex items-center gap-2 border border-purple-100";
              } else {
                contentClass += " text-slate-700 leading-relaxed";
              }

              return (
                <div 
                  key={block.id} 
                  className={containerClass}
                  onClick={() => setSelectedId(block.id)}
                >
                  <div className={`absolute -left-10 top-1.5 p-1 text-slate-300 cursor-grab active:cursor-grabbing hover:text-slate-500 transition-opacity ${isSelected ? 'opacity-100' : 'opacity-0 group-hover:opacity-100'}`}>
                    <GripVertical size={20} />
                  </div>

                  <div className={`absolute -right-24 top-2 text-xs font-mono text-slate-400 flex items-center gap-1 transition-opacity ${isSelected ? 'opacity-100' : 'opacity-0 group-hover:opacity-100'}`}>
                    <Hash size={12} />
                    {block.id}
                  </div>

                  <div className="relative pl-3">
                    {sideBorder}
                    <div 
                      contentEditable 
                      suppressContentEditableWarning
                      className={contentClass}
                    >
                      {/* [新增] 如果是流程链接，显示特定内容结构 */}
                      {block.type === 'process_link' ? (
                        <>
                           <GitGraph size={18} className="text-purple-500 flex-shrink-0"/>
                           <span>{block.content}</span>
                           <span className="text-xs bg-purple-200 text-purple-700 px-2 py-0.5 rounded-full ml-auto">
                             跳转
                           </span>
                        </>
                      ) : (
                        <>
                          {block.type !== 'heading' && <span className="font-mono text-slate-400 mr-2 select-none">{block.number}</span>}
                          {block.content}
                        </>
                      )}
                    </div>
                  </div>
                  
                </div>
              );
            })}
          </div>
        </div>
      </div>

      {/* 3. 右侧栏 */}
      <div className="w-80 bg-white border-l border-slate-200 flex flex-col shadow-xl z-20">
        <div className="h-14 border-b border-slate-100 flex items-center px-4 font-semibold text-slate-700 bg-slate-50">
          <Settings size={18} className="mr-2 text-slate-500" />
          条款属性
        </div>

        {selectedBlock ? (
          <div className="flex-1 overflow-y-auto p-4 space-y-6">
            
            <div className="space-y-3">
              <label className="text-xs font-bold text-slate-400 uppercase tracking-wider">数据定义</label>
              
              <div className="space-y-1">
                <span className="text-xs text-slate-500">条款 ID</span>
                <div className="flex items-center justify-between bg-slate-100 p-2 rounded text-sm font-mono text-slate-600">
                  {selectedBlock.id}
                  <button className="text-blue-500 hover:text-blue-600"><LinkIcon size={14}/></button>
                </div>
              </div>

              <div className="space-y-1">
                <span className="text-xs text-slate-500">语义类型</span>
                <select 
                  className="w-full p-2 border border-slate-200 rounded text-sm bg-white focus:ring-2 focus:ring-blue-500 outline-none"
                  value={selectedBlock.type}
                  onChange={(e) => updateBlock(selectedBlock.id, 'type', e.target.value)}
                >
                  <option value="clause">普通条款 (Clause)</option>
                  <option value="rule">控制规则 (Rule)</option>
                  <option value="risk">风险点 (Risk)</option>
                  <option value="process_link">流程挂载 (Link)</option> {/* [新增] 选项 */}
                  <option value="heading">标题 (Heading)</option>
                </select>
              </div>
            </div>

            <hr className="border-slate-100" />

            <div className="space-y-3">
              <label className="text-xs font-bold text-slate-400 uppercase tracking-wider">业务元数据</label>
              
              {/* [新增] 针对新模块的特定属性表单 */}
              {selectedBlock.type === 'process_link' && (
                <div className="p-3 bg-purple-50 rounded border border-purple-100 space-y-2">
                  <div className="flex items-center gap-2 text-purple-700 font-medium text-sm">
                    <GitGraph size={16} /> 流程配置
                  </div>
                  <div className="space-y-1">
                    <label className="text-xs text-purple-600">目标流程ID</label>
                    <input 
                       type="text" 
                       defaultValue={selectedBlock.meta.targetProcessId || 'BPMN-XXX'}
                       className="w-full px-2 py-1 text-xs border border-purple-200 rounded bg-white"
                    />
                  </div>
                </div>
              )}

              {selectedBlock.type === 'risk' && (
                <div className="p-3 bg-amber-50 rounded border border-amber-100 space-y-2">
                  <div className="flex items-center gap-2 text-amber-700 font-medium text-sm">
                    <AlertTriangle size={16} /> 风险管控
                  </div>
                  <div className="text-xs text-amber-600">此条款已被标记为高风险点，需要关联相应的控制措施。</div>
                </div>
              )}

              {selectedBlock.type !== 'process_link' && (
                <>
                  <div className="space-y-1">
                    <span className="text-xs text-slate-500">责任岗位</span>
                    <div className="flex items-center gap-2 p-2 border border-slate-200 rounded text-sm hover:border-blue-300 cursor-pointer bg-white">
                      <User size={14} className="text-slate-400" />
                      {selectedBlock.meta.owner || '未分配'}
                    </div>
                  </div>
                  
                  <div className="space-y-1">
                    <span className="text-xs text-slate-500">状态</span>
                    <div className="flex gap-2">
                      <span className="px-2 py-1 bg-green-100 text-green-700 rounded text-xs font-medium">生效中</span>
                      {selectedBlock.subtype === 'mandatory' && <span className="px-2 py-1 bg-red-100 text-red-700 rounded text-xs font-medium">强制性</span>}
                    </div>
                  </div>
                </>
              )}
            </div>

            <hr className="border-slate-100" />

            {/* 关联分析 */}
            <div className="space-y-3">
              <label className="text-xs font-bold text-slate-400 uppercase tracking-wider flex justify-between">
                引用分析 (Impact)
                <span className="bg-blue-100 text-blue-700 px-1.5 rounded-full text-xs">
                  {selectedBlock.meta.backlinks || 0}
                </span>
              </label>

              {selectedBlock.meta.backlinks ? (
                 <div className="space-y-2">
                   <div className="text-xs text-slate-500 mb-2">以下流程直接引用了该条款：</div>
                   <div className="flex items-center gap-2 p-2 bg-slate-50 rounded border border-slate-200 hover:border-blue-300 cursor-pointer transition-colors group">
                      <Activity size={14} className="text-blue-500" />
                      <div className="flex-1">
                        <div className="text-xs font-medium text-slate-700 group-hover:text-blue-600">财务报销流程.bpmn</div>
                        <div className="text-[10px] text-slate-400">节点: 部门经理审批</div>
                      </div>
                      <ChevronRight size={14} className="text-slate-300" />
                   </div>
                 </div>
              ) : (
                <div className="text-sm text-slate-400 italic text-center py-4">
                  暂无关联引用
                </div>
              )}
            </div>

          </div>
        ) : (
          <div className="flex-1 flex flex-col items-center justify-center text-slate-400 p-8 text-center">
            <Layout size={40} className="mb-4 opacity-20" />
            <p className="text-sm">在左侧选择一个段落以查看属性</p>
          </div>
        )}
      </div>
    </div>
  );
}
